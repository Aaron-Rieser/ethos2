function showFeed() { if (!userLocation) { alert('Please wait for your location to be determined'); return; } const radius = document.querySelector('.range-slider').value; // Store the selected location and radius localStorage.setItem('selectedLat', userLocation[0].toFixed(8)); localStorage.setItem('selectedLng', userLocation[1].toFixed(8)); localStorage.setItem('selectedRadius', radius); // Add precision to coordinates const params = new URLSearchParams({ lat: userLocation[0].toFixed(8), lng: userLocation[1].toFixed(8), radius: radius }); console.log('Redirecting with parameters:', params.toString()); window.location.href = `index.html?${params.toString()}`; }
